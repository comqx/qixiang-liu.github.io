问题描述

这里先回想下变量是什么时候才申请内存？

func main() {
	m := make(map[string]*student)
	stus := []student{
		{name: "小王子", age: 18},
		{name: "娜扎", age: 23},
		{name: "大王八", age: 9000},
	}

	for _, stu := range stus {
		fmt.Println(stu)
		fmt.Printf("%p,%v\n", &stu, stu) //这里地址相同，值不同？？？
		m[stu.name] = &stu
	}
	for k, v := range m {
		fmt.Println(k, "=>", v.name)
	}
}

// 结果：
{小王子 18}
0xc00000c040,{小王子 18}  //地址相同，值不同，相等于做了赋值的操作
{娜扎 23}
0xc00000c040,{娜扎 23}
{大王八 9000}
0xc00000c040,{大王八 9000}
小王子 => 大王八
娜扎 => 大王八
大王八 => 大王八
